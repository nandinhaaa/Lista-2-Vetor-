apresentar o vetor juntamente com:
• melhor tempo;

• tempo médio das N voltas para cada piloto*/

#include<math.h>

int main(){
float tmedio;

for(i=0;i<10;i++)
printf("\ndigite o tempo obtido na %d volta em minutos: ",i);
scanf("%d",&tempo[i]);
if( tempo[i] > mt)
{
mt=tempo[i];
melhorvolta=i;
}
st=st+tempo[i];

}
